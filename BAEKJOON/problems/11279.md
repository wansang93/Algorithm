```python
import heapq
from sys import stdin

N = int(stdin.readline())
heap = []
heapq.heapify(heap)
for _ in range(N):
    num = int(stdin.readline())
    if num == 0:
        if heap:
            max_num = heapq.heappop(heap)[1]
            print(max_num)
        else:
            print(0)
    else:
        heapq.heappush(heap, (-num, num))

```

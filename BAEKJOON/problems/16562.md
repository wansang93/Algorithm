import sys
input = sys.stdin.readline

def find(x):
    if x != parent[x]:
        parent[x] = find(parent[x])
    return parent[x]


def union(x, y):
    x, y = find(x), find(y)
    if lst[x-1] <= lst[y-1]:
        parent[y] = x
    else:
        parent[x] = y


N, M, k = map(int, input().split())
lst = list(map(int, input().split()))
lst2 = [list(map(int, input().split())) for _ in range(M)]

parent = [i for i in range(N+1)]

for v1, v2 in lst2:
    union(v1, v2)

groups = dict()
for i in range(1, N+1):
    root = find(i)
    if root not in groups:
        groups[root] = lst[i-1]
    else:
        groups[root] = min(groups[root], lst[i-1])

cost = sum(groups.values())

if cost <= k:
    print(cost)
else:
    print("Oh no")

나중에 다시 풀어보기

## Python

고수의 풀이

```python
def concatenate(r1, r2):
    return [''.join(x) for x in zip(r1, r2, r1)]
 
def star10(n):
    if n == 1:
        return ['*']
    n //= 3
    x = star10(n)
    a = concatenate(x, x)
    b = concatenate(x, [' '*n]*n)
 
    return a + b + a

print('\n'.join(star10(int(input()))))
```

시간 초과 에러

```python
N = int(input())
result = N
base = 3
E = 0
while 1 < result:
    result //= base
    E += 1

star = [['*'] * N for _ in range(N)]

def star3(x, y, e):
    global star, E
    if e > E:
        return
    if (x//3**e) % 3 == 1 and (y//3**e) % 3 == 1:
        star[x][y] = ' '
        return
    star3(x, y, e+1)

for s in star:
    print(''.join(s))
```

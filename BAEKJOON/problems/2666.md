import sys
input = sys.stdin.readline


def dfs(depth, l, r):
    # depth가 M일때 까지
    if depth == M:
        return 0
    
    # 방문했으면 방문한 dp 체크
    if dp[depth][l][r] != -1:
        return dp[depth][l][r]
        
    v = lst[depth]
    left = abs(l-v) + dfs(depth+1, v, r)
    right = abs(r-v) + dfs(depth+1, l, v)
    dp[depth][l][r] = min(left, right)
    
    return dp[depth][l][r]


N = int(input())
a, b = map(int, input().split())
M = int(input())
lst = [int(input()) for _ in range(M)]

dp = [[[-1] * (N+1) for _ in range(N+1)] for _ in range(M+1)]
ans = dfs(0, a, b)

print(ans)

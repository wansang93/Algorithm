import sys
input = sys.stdin.readline

N, M, B = map(int, input().split())
lst = [list(map(int, input().split())) for _ in range(N)]
time_ans = 10e12
target_ans = 0

for target in range(257):
    inven = B
    time = 0
    for i in range(N):
        for j in range(M):
            if target > lst[i][j]:
                put = target - lst[i][j]
                time += put
                inven -= put
            elif target < lst[i][j]:
                get = lst[i][j] - target
                time += get * 2
                inven += get
    if inven < 0:
        break
    if time <= time_ans:
        time_ans = time
        target_ans = target

print(time_ans, target_ans)

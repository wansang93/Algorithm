from collections import deque
from itertools import permutations

PADDING_SIZE = 3
MAX_SIZE = 60
INF = 10**9

N = int(input())
lst = list(map(int, input().split()))
lst += [0] * (PADDING_SIZE-N)

dp = [[[INF] * (MAX_SIZE+1) for _ in range(MAX_SIZE+1)] for _ in range(MAX_SIZE+1)]
per = list(permutations([9, 3, 1], 3))

q = deque([lst])
dp[lst[0]][lst[1]][lst[2]] = 0
while q:
    x, y, z = q.popleft()
    if x == 0 and y == 0 and z == 0:
        break

    for a, b, c in per:
        a = max(0, x-a)
        b = max(0, y-b)
        c = max(0, z-c)
        if dp[a][b][c] != INF:
            continue

        dp[a][b][c] = dp[x][y][z] + 1
        q.append((a, b, c))

print(dp[0][0][0])

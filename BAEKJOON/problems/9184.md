```python
import sys
input = sys.stdin.readline


N = 21
lst = [[[-1] * N for _ in range(N)] for _ in range(N)]


def w(a, b, c):
    if a <= 0 or b <= 0 or c <= 0:
        return 1

    if a > 20 or b > 20 or c > 20:
        return w(20, 20, 20)

    if lst[a][b][c] != -1:
        return lst[a][b][c]
    elif a < b and b < c:
        lst[a][b][c] = w(a, b, c-1) + w(a, b-1, c-1) - w(a, b-1, c)
    else:
        lst[a][b][c] = w(a-1, b, c) + w(a-1, b-1, c) + w(a-1, b, c-1) - w(a-1, b-1, c-1)

    return lst[a][b][c]


while True:
    a, b, c = map(int, input().split())
    if a == b == c == -1:
        break
    
    ans = w(a, b, c)
    print(f"w({a}, {b}, {c}) = {ans}")

```

```python
"""
1: x
2: y
3: x+y
4: x+2y
5: 2x+3y
6: 3x+5y
7: 5x+8y
...
dp수 : 1 1 2 3 5 '8' 13...
y계수: 0 1 1 2 3 '5' 8 ...
x계수: 1 0 1 1 2 '3' 5 ...
날짜 : 1 2 3 4 5 '6' 7
3x + 5y = K
K - 3i % 5 == 0:
    i, K-3i // 5
"""

D, K = map(int, input().split())
dp = [0] * (D+1)
dp[1] = 1
for i in range(2, D+1):
    dp[i] = dp[i-1] + dp[i-2]

day1 = dp[D-2]
day2 = dp[D-1]
cnt1 = 0
cnt2 = 0
for i in range(K):
    if (K - day1 * i) % day2 == 0:
        cnt1 = i
        cnt2 = (K- day1 * i) // day2
        break

print(cnt1)
print(cnt2)
```

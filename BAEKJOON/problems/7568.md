```python
from sys import stdin

N = int(stdin.readline())
lst = []
for _ in range(N):
    weight, height = stdin.readline().strip().split()
    lst.append((weight, height))

for i in lst:
    rank = 1
    for j in lst:
        if i[0] < j[0] and i[1] < j[1]:
            rank += 1
    print(rank, end=' ')

```

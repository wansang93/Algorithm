# pypy ì œì¶œ, O(NM)
import sys
sys.setrecursionlimit(10**5)
input = sys.stdin.readline

def dfs(now, depth):
    visit[now] = 1
    dis[now] = depth
    for nxt in graph[now]:
        if visit[nxt]:
            continue
        parent[nxt] = now
        dfs(nxt, depth+1)

def lca(a, b):
    while dis[a] != dis[b]:
        if dis[a] > dis[b]:
            a = parent[a]
        else:
            b = parent[b]
    
    while a != b:
        a = parent[a]
        b = parent[b]
    
    return a

N = int(input())
graph = [[] for _ in range(N+1)]
for _ in range(N-1):
    a, b = map(int, input().split())
    graph[a].append(b)
    graph[b].append(a)

parent = [0] * (N+1)
dis = [0] * (N+1)
visit = [0] * (N+1)

dfs(1, 0)

M = int(input())
for _ in range(M):
    a, b = map(int, input().split())
    print(lca(a, b))

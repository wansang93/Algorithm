```python
import sys
input = sys.stdin.readline

N, M = map(int, input().split())
activates = [0] + list(map(int, input().split()))
costs = [0] + list(map(int, input().split()))

SIZE = sum(costs)
dp = [[0] * (SIZE+1) for _ in range(N+1)]

ans = SIZE
for i in range(1, N+1):
    activate, cost = activates[i], costs[i]
    for j in range(1, SIZE+1):
        if j < cost:
            dp[i][j] = dp[i-1][j]
        else:
            dp[i][j] = max(dp[i-1][j], activate + dp[i-1][j-cost])
        if dp[i][j] >= M:
            ans = min(ans, j)

print(ans if M != 0 else 0)

```

s = input()
answer = 0

for i in range(len(s)):
    l, r = i, i
    while l >= 0 and r < len(s) and s[l] == s[r]:
        answer = max(r-l+1, answer)
        l -= 1
        r += 1
    
    l, r = i, i+1
    while l >= 0 and r < len(s) and s[l] == s[r]:
        answer = max(r-l+1, answer)
        l -= 1
        r += 1

print(answer)

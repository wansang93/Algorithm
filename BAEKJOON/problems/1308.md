import sys, datetime
input = sys.stdin.readline

ny, nm, nd = map(int, input().split())
dy, dm, dd = map(int, input().split())

n_day = datetime.date(ny, nm, nd)
d_day = datetime.date(dy, dm, dd)
day_left = d_day - n_day
if n_day <= datetime.date(dy-1000, dm, dd):
    print("gg")
else:
    ans = day_left.days
    print(f"D-{ans}")

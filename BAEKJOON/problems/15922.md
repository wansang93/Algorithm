```python
import sys

input = sys.stdin.readline

N = int(input())
lst = [tuple(map(int, input().split())) for _ in range(N)]
lst.sort()

s, e = lst[0]
ans = 0
for ns, ne in lst[1:]:
    if ns <= e:
        if ne > e:
            e = ne
    else:
        ans += e - s
        s, e = ns, ne

ans += e - s
print(ans)

```

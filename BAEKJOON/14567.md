input = sys.stdin.readline

def topology_sort():
    q = deque([])
    for i in range(1, N+1):

            q.append(i)
    
    cnt = 1
    ret = [1] * (N+1)
    while q:
        for _ in range(len(q)):
            now = q.popleft()
            ret[now] = cnt
            for nxt in graph[now]:
                indegree[nxt] -= 1
                if not indegree[nxt]:
                    q.append(nxt)
        cnt += 1
    
    return ret[1:]

N, M = map(int, input().split())
graph = [[] for _ in range(N+1)]
indegree = [0] * (N+1)

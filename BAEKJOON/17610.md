```python
# 내 풀이
k = int(input())
lst = list(map(int, input().split()))

visit = set([0])
for i in range(k):
    tmp = set()
    for v in visit:
        tmp.add(v+lst[i])
        tmp.add(abs(v - lst[i]))
    visit |= tmp

ans = sum(lst) - len(visit) + 1
print(ans)

```

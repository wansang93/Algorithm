# pypy 제출
import sys
input = sys.stdin.readline

S1 = input().strip()
S2 = input().strip()

N, M = len(S1), len(S2)
dp = [[0] * (M+1) for _ in range(N+1)]

ans = 0
for i in range(N):
    for j in range(M):
        if S1[i] == S2[j]:
            dp[i+1][j+1] = dp[i][j] + 1
            ans = max(ans, dp[i+1][j+1])

print(ans)
